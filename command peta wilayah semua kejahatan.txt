trains.map <- function(categories, n) {

    new.train <- filter(train, Category %in% categories) %>% droplevels() 
    
    pal <- colorFactor(brewer.pal(length(unique(new.train$Category)), "Set3"),  #Set3 kayaknya paling banyak variasi warna
                       domain = new.train$Category)

    leaflet(new.train[1:n,]) %>% 
        addProviderTiles("CartoDB.Positron") %>% 
        addCircleMarkers (lng =  ~X, lat =  ~Y, 
                          color = ~pal(Category),
                          opacity = .7, radius  = 1) %>%  
        addLegend(pal = pal, values = new.train$Category)
}
